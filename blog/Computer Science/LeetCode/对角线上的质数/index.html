    class Solution {
public:
    int diagonalPrime(vector<vector<int>>& nums) {
        int m = nums.size(); // è¡Œ
        int max = 0;
        for(int i=0;i<m;i++){
            if(is_zhisu(nums[i][i]) & nums[i][i]>max){
                max = nums[i][i];
            }
            if(is_zhisu(nums[i][m-i-1]) & nums[i][m-i-1]>max){
                max = nums[i][m-i-1];
            }
        }
        return max;
    }
    bool is_zhisu(int n){
        for(int i=2;i<int(n/2);i++){
            if(n%i==0){
                return false;
            }
        }
        return true;
    }
};
